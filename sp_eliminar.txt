DELIMITER $$
CREATE DEFINER=`root`@`localhost` PROCEDURE `SP_ELIMINAR_RESULTADOS`(IN `idSeguimientoR` INT(11))
    NO SQL
BEGIN 

DELETE FROM `tab_resultado_ficha` WHERE `Seguimiento_idSeguimiento`=idSeguimientoR;

DELETE FROM `tab_resultado_pie` WHERE `Seguimiento_idSeguimiento`=idSeguimientoR;

DELETE FROM `tab_seguimiento` WHERE `idSeguimiento`=idSeguimientoR;


END$$
DELIMITER ;
